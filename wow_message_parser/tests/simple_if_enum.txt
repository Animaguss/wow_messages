#[cfg(any(feature = "async_tokio", feature = "async_std"))]
use async_trait::async_trait;

use tokio::io::{AsyncReadExt, AsyncWriteExt};
#[cfg(feature = "async_std")]
use async_std::io::{ReadExt, WriteExt};

#[derive(Debug, PartialEq, Clone, Default)]
pub struct SimpleIfEnum {
    pub enum: SimpleIfEnumSimpleEnum,
}

#[cfg_attr(any(feature = "async_tokio", feature = "async_std"), async_trait)]
impl ReadableAndWritable for SimpleIfEnum {
    type Error = SimpleIfEnumError;

    #[cfg(feature = "sync")]
    fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, Self::Error> {
        // enum: SimpleEnum
        let enum = SimpleEnum::read(r)?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "sync")]
    fn write<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        self.enum.write(w)?;

        match &self.enum {
            SimpleIfEnumSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumSimpleEnum::ONE => {}
            SimpleIfEnumSimpleEnum::TWO => {}
        }

        Ok(())
    }

    #[cfg(feature = "async_tokio")]
    async fn tokio_read<R: AsyncReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, Self::Error> {
        // enum: SimpleEnum
        let enum = SimpleEnum::tokio_read(r).await?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "async_tokio")]
    async fn tokio_write<W: AsyncWriteExt + Unpin + Send>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        self.enum.tokio_write(w).await?;

        match &self.enum {
            SimpleIfEnumSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes()).await?;

            }
            SimpleIfEnumSimpleEnum::ONE => {}
            SimpleIfEnumSimpleEnum::TWO => {}
        }

        Ok(())
    }

    #[cfg(feature = "async_std")]
    async fn astd_read<R: ReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, Self::Error> {
        // enum: SimpleEnum
        let enum = SimpleEnum::astd_read(r).await?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "async_std")]
    async fn astd_write<W: WriteExt + Unpin + Send>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        self.enum.astd_write(w).await?;

        match &self.enum {
            SimpleIfEnumSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes()).await?;

            }
            SimpleIfEnumSimpleEnum::ONE => {}
            SimpleIfEnumSimpleEnum::TWO => {}
        }

        Ok(())
    }

}

impl VariableSized for SimpleIfEnum {
    fn size(&self) -> usize {
        self.enum.size() // enum: SimpleEnum and subfields
    }
}

impl MaximumPossibleSized for SimpleIfEnum {
    fn maximum_possible_size() -> usize {
        SimpleEnum::maximum_possible_size() // enum: SimpleEnum
    }
}

#[derive(Debug)]
pub enum SimpleIfEnumError {
    Io(std::io::Error),
    SimpleEnum(SimpleEnumError),
}

impl std::error::Error for SimpleIfEnumError {}
impl std::fmt::Display for SimpleIfEnumError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::SimpleEnum(i) => i.fmt(f),
        }
    }
}

impl From<std::io::Error> for SimpleIfEnumError {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<SimpleEnumError> for SimpleIfEnumError {
    fn from(e: SimpleEnumError) -> Self {
        Self::SimpleEnum(e)
    }
}

#[derive(Debug, PartialEq, Clone)]
pub enum SimpleIfEnumSimpleEnum {
    ZERO {
        basic: u8,
    },
    ONE,
    TWO,
}

impl From<&SimpleEnum> for SimpleIfEnumSimpleEnum {
    fn from(e: &SimpleEnum) -> Self {
        match &e {
            SimpleEnum::ZERO => Self::ZERO {
                basic: Default::default(),
            },
            SimpleEnum::ONE => Self::ONE,
            SimpleEnum::TWO => Self::TWO,
        }
    }
}

impl From<&SimpleIfEnumSimpleEnum> for SimpleEnum {
    fn from(v: &SimpleIfEnumSimpleEnum) -> Self {
        match &v {
            SimpleIfEnumSimpleEnum::ZERO { .. } => Self::ZERO,
            SimpleIfEnumSimpleEnum::ONE => Self::ONE,
            SimpleIfEnumSimpleEnum::TWO => Self::TWO,
        }
    }
}

impl Default for SimpleIfEnumSimpleEnum {
    fn default() -> Self {
        // First enumerator without any fields
        Self::ZERO {
            basic: Default::default(),
        }
    }
}
