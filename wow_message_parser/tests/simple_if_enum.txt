use std::convert::{TryFrom, TryInto};
use crate::world::v1::v12::SimpleEnum;
#[cfg(feature = "tokio")]
use tokio::io::AsyncReadExt;
#[cfg(feature = "async-std")]
use async_std::io::ReadExt;
use std::io::Write;

#[derive(Debug, PartialEq, Clone, Default)]
pub struct SimpleIfEnum {
    pub enum: SimpleIfEnumSimpleEnum,
}

impl SimpleIfEnum {
    pub(crate) fn as_bytes(&self) -> Result<Vec<u8>, std::io::Error> {
        let mut w = Vec::with_capacity(self.size());
        // enum: SimpleEnum
        w.write_all(&(self.enum.as_int() as u8).to_le_bytes())?;

        match &self.enum {
            SimpleIfEnumSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumSimpleEnum::ONE => {}
            SimpleIfEnumSimpleEnum::TWO => {}
        }

        Ok(w)
    }
}

impl SimpleIfEnum {
    #[cfg(feature = "sync")]
    pub(crate) fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::read_u8_le(r)?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "tokio")]
    pub(crate) async fn tokio_read<R: AsyncReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::tokio_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "async-std")]
    pub(crate) async fn astd_read<R: ReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::astd_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => SimpleIfEnumSimpleEnum::ONE,
            SimpleEnum::TWO => SimpleIfEnumSimpleEnum::TWO,
        };

        Ok(Self {
            enum: enum_if,
        })
    }

}

impl SimpleIfEnum {
    pub fn size(&self) -> usize {
        0
        + self.enum.size() // enum: SimpleIfEnumSimpleEnum
    }
}

#[derive(Debug)]
pub enum SimpleIfEnumError {
    Io(std::io::Error),
    Enum(crate::errors::EnumError),
}

impl std::error::Error for SimpleIfEnumError {}
impl std::fmt::Display for SimpleIfEnumError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::Enum(e) => e.fmt(f),
        }
    }
}

impl From<std::io::Error> for SimpleIfEnumError {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<crate::errors::EnumError> for SimpleIfEnumError {
    fn from(e: crate::errors::EnumError) -> Self {
        Self::Enum(e)
    }
}

#[derive(Debug, PartialEq, Clone)]
pub enum SimpleIfEnumSimpleEnum {
    ZERO {
        basic: u8,
    },
    ONE,
    TWO,
}

impl Default for SimpleIfEnumSimpleEnum {
    fn default() -> Self {
        // First enumerator without any fields
        Self::ZERO {
            basic: Default::default(),
        }
    }
}

impl SimpleIfEnumSimpleEnum {
    pub(crate) const fn as_int(&self) -> u8 {
        match self {
            Self::ZERO { .. } => 0,
            Self::ONE => 1,
            Self::TWO => 2,
        }
    }

}

impl SimpleIfEnumSimpleEnum {
    pub fn size(&self) -> usize {
        match self {
            Self::ZERO {
                basic,
            } => {
                1
                + 1 // basic: u8
            }
            Self::ONE => {
                1
            }
            Self::TWO => {
                1
            }
        }
    }
}
