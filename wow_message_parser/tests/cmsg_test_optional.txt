        let c_string = String::from_utf8(c_string)?;

        // b_u8: u8

        // optional server

                optional_u8,

            // optional_u8: u8

impl VariableSized for CMSG_TEST_OPTIONAL {
    fn size(&self) -> usize {
        self.c_string.len() + 1 // c_string: CString and Null Terminator
        + 1 // b_u8: u8
        + {
            if let Some(v) = &self.server {
                v.size()
            } else {
                0
            }
        } // optional server
    }
}
