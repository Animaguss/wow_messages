use crate::{ClientMessageWrite, MessageBody};
use wow_srp::header_crypto::Encrypter;

#[cfg(any(feature = "async_tokio", feature = "async_std"))]
use crate::AsyncReadWrite;
#[cfg(any(feature = "async_tokio", feature = "async_std"))]
use async_trait::async_trait;

#[cfg(feature = "async_std")]
use async_std::io::{ReadExt, WriteExt};

#[derive(Debug, PartialEq, Clone, Default)]
pub struct CMSG_TEST_ENDLESS_U8 {
    pub c_string: String,
    pub b_u8: u8,
    pub endless: Vec<u8>,
}

impl ClientMessageWrite for CMSG_TEST_ENDLESS_U8 {
    const OPCODE: u16 = 0x1337;

    fn size_without_size_or_opcode_fields(&self) -> u16 {
        self.size() as u16
    }

}

impl MessageBody for CMSG_TEST_ENDLESS_U8 {
    type Error = CMSG_TEST_ENDLESS_U8Error;

    fn read_body<R: std::io::Read>(r: &mut R, body_size: u32) -> std::result::Result<Self, Self::Error> {
        // c_string: CString
        let c_string = crate::util::read_c_string_to_vec(r)?;
        let c_string = String::from_utf8(c_string)?;

        // b_u8: u8
        let b_u8 = crate::util::read_u8_le(r)?;

        // endless: u8[-]
        let mut current_size = {
            c_string.len() + 1 // c_string: CString and Null Terminator
            + 1 // b_u8: u8
        };
        let mut endless = Vec::with_capacity(body_size as usize - current_size);
        while current_size < (body_size as usize) {
            endless.push(crate::util::read_u8_le(r)?);
            current_size += 1;
        }

        Ok(Self {
            c_string,
            b_u8,
            endless,
        })
    }

    fn write_body<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // c_string: CString
        w.write_all(self.c_string.as_bytes())?;
        // Null terminator
        w.write_all(&[0])?;

        // b_u8: u8
        w.write_all(&self.b_u8.to_le_bytes())?;

        // endless: u8[-]
        for i in self.endless.iter() {
            w.write_all(&i.to_le_bytes())?;
        }

        Ok(())
    }
}

impl VariableSized for CMSG_TEST_ENDLESS_U8 {
    fn size(&self) -> usize {
        self.c_string.len() + 1 // c_string: CString and Null Terminator
        + 1 // b_u8: u8
        + self.endless.len() * core::mem::size_of::<u8>() // endless: u8[-]
    }
}

impl MaximumPossibleSized for CMSG_TEST_ENDLESS_U8 {
    fn maximum_possible_size() -> usize {
        256 // c_string: CString
        + 1 // b_u8: u8
        + 65536 // endless: u8[-]
    }
}

#[derive(Debug)]
pub enum CMSG_TEST_ENDLESS_U8Error {
    Io(std::io::Error),
    String(std::string::FromUtf8Error),
}

impl std::error::Error for CMSG_TEST_ENDLESS_U8Error {}
impl std::fmt::Display for CMSG_TEST_ENDLESS_U8Error {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::String(i) => i.fmt(f),
        }
    }
}

impl From<std::io::Error> for CMSG_TEST_ENDLESS_U8Error {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<std::string::FromUtf8Error> for CMSG_TEST_ENDLESS_U8Error {
    fn from(e: std::string::FromUtf8Error) -> Self {
        Self::String(e)
    }
}
