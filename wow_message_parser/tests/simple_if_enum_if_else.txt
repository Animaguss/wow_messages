use std::convert::{TryFrom, TryInto};
use crate::world::v1::v12::{SimpleEnum, SimpleEnumError};
#[cfg(feature = "tokio")]
use tokio::io::AsyncReadExt;
#[cfg(feature = "async-std")]
use async_std::io::ReadExt;
use std::io::Write;

#[derive(Debug, PartialEq, Clone, Default)]
pub struct SimpleIfEnumIfElse {
    pub enum: SimpleIfEnumIfElseSimpleEnum,
}

impl SimpleIfEnumIfElse {
    pub(crate) fn as_bytes(&self) -> Result<Vec<u8>, std::io::Error> {
        let mut w = Vec::with_capacity(self.size());
        // enum: SimpleEnum
        w.write_all(&(self.enum.as_int() as u8).to_le_bytes())?;

        match &self.enum {
            SimpleIfEnumIfElseSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumIfElseSimpleEnum::ONE {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumIfElseSimpleEnum::TWO {
                other,
            } => {
                // other: u16
                w.write_all(&other.to_le_bytes())?;

            }
        }

        Ok(w)
    }
}

impl SimpleIfEnumIfElse {
    #[cfg(feature = "sync")]
    pub(crate) fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumIfElseError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::read_u8_le(r)?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::ONE {
                    basic,
                }
            }
            SimpleEnum::TWO => {
                // other: u16
                let other = crate::util::read_u16_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::TWO {
                    other,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "tokio")]
    pub(crate) async fn tokio_read<R: AsyncReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumIfElseError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::tokio_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => {
                // basic: u8
                let basic = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::ONE {
                    basic,
                }
            }
            SimpleEnum::TWO => {
                // other: u16
                let other = crate::util::tokio_read_u16_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::TWO {
                    other,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    #[cfg(feature = "async-std")]
    pub(crate) async fn astd_read<R: ReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumIfElseError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::astd_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => {
                // basic: u8
                let basic = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::ONE {
                    basic,
                }
            }
            SimpleEnum::TWO => {
                // other: u16
                let other = crate::util::astd_read_u16_le(r).await?;

                SimpleIfEnumIfElseSimpleEnum::TWO {
                    other,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

}

impl SimpleIfEnumIfElse {
    pub fn size(&self) -> usize {
        0
        + self.enum.size() // enum: SimpleIfEnumIfElseSimpleEnum
    }
}

#[derive(Debug)]
pub enum SimpleIfEnumIfElseError {
    Io(std::io::Error),
    SimpleEnum(SimpleEnumError),
}

impl std::error::Error for SimpleIfEnumIfElseError {}
impl std::fmt::Display for SimpleIfEnumIfElseError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::SimpleEnum(i) => i.fmt(f),
        }
    }
}

impl From<std::io::Error> for SimpleIfEnumIfElseError {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<SimpleEnumError> for SimpleIfEnumIfElseError {
    fn from(e: SimpleEnumError) -> Self {
        Self::SimpleEnum(e)
    }
}

#[derive(Debug, PartialEq, Clone)]
pub enum SimpleIfEnumIfElseSimpleEnum {
    ZERO {
        basic: u8,
    },
    ONE {
        basic: u8,
    },
    TWO {
        other: u16,
    },
}

impl Default for SimpleIfEnumIfElseSimpleEnum {
    fn default() -> Self {
        // First enumerator without any fields
        Self::ZERO {
            basic: Default::default(),
        }
    }
}

impl SimpleIfEnumIfElseSimpleEnum {
    pub(crate) const fn as_int(&self) -> u8 {
        match self {
            Self::ZERO { .. } => 0,
            Self::ONE { .. } => 1,
            Self::TWO { .. } => 2,
        }
    }

}

impl SimpleIfEnumIfElseSimpleEnum {
    pub fn size(&self) -> usize {
        match self {
            Self::ZERO {
                basic,
            } => {
                1
                + 1 // basic: u8
            }
            Self::ONE {
                basic,
            } => {
                1
                + 1 // basic: u8
            }
            Self::TWO {
                other,
            } => {
                1
                + 2 // other: u16
            }
        }
    }
}
