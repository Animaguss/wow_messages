use std::convert::{TryFrom, TryInto};
use crate::world::v1::v12::{SimpleEnum, SimpleEnumError};
use crate::{ConstantSized, MaximumPossibleSized, ReadableAndWritable, VariableSized};

#[derive(Debug, PartialEq, Clone, Default)]
pub struct SimpleIfEnumIfElse {
    pub enum: SimpleIfEnumIfElseSimpleEnum,
}

impl ReadableAndWritable for SimpleIfEnumIfElse {
    type Error = SimpleIfEnumIfElseError;

    fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, Self::Error> {
        // enum: SimpleEnum
        let enum = SimpleEnum::read(r)?;

        let enum_if = match enum {
            SimpleEnum::ZERO => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::ZERO {
                    basic,
                }
            }
            SimpleEnum::ONE => {
                // basic: u8
                let basic = crate::util::read_u8_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::ONE {
                    basic,
                }
            }
            SimpleEnum::TWO => {
                // other: u16
                let other = crate::util::read_u16_le(r)?;

                SimpleIfEnumIfElseSimpleEnum::TWO {
                    other,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    fn write<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        self.enum.write(w)?;

        match &self.enum {
            SimpleIfEnumIfElseSimpleEnum::ZERO {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumIfElseSimpleEnum::ONE {
                basic,
            } => {
                // basic: u8
                w.write_all(&basic.to_le_bytes())?;

            }
            SimpleIfEnumIfElseSimpleEnum::TWO {
                other,
            } => {
                // other: u16
                w.write_all(&other.to_le_bytes())?;

            }
        }

        Ok(())
    }

}

impl VariableSized for SimpleIfEnumIfElse {
    fn size(&self) -> usize {
        self.enum.size() // enum: SimpleEnum and subfields
    }
}

impl MaximumPossibleSized for SimpleIfEnumIfElse {
    fn maximum_possible_size() -> usize {
        SimpleEnum::maximum_possible_size() // enum: SimpleEnum
    }
}

#[derive(Debug)]
pub enum SimpleIfEnumIfElseError {
    Io(std::io::Error),
    SimpleEnum(SimpleEnumError),
}

impl std::error::Error for SimpleIfEnumIfElseError {}
impl std::fmt::Display for SimpleIfEnumIfElseError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::SimpleEnum(i) => i.fmt(f),
        }
    }
}

impl From<std::io::Error> for SimpleIfEnumIfElseError {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<SimpleEnumError> for SimpleIfEnumIfElseError {
    fn from(e: SimpleEnumError) -> Self {
        Self::SimpleEnum(e)
    }
}

#[derive(Debug, PartialEq, Clone)]
pub enum SimpleIfEnumIfElseSimpleEnum {
    ZERO {
        basic: u8,
    },
    ONE {
        basic: u8,
    },
    TWO {
        other: u16,
    },
}

impl From<&SimpleEnum> for SimpleIfEnumIfElseSimpleEnum {
    fn from(e: &SimpleEnum) -> Self {
        match &e {
            SimpleEnum::ZERO => Self::ZERO {
                basic: Default::default(),
            },
            SimpleEnum::ONE => Self::ONE {
                basic: Default::default(),
            },
            SimpleEnum::TWO => Self::TWO {
                other: Default::default(),
            },
        }
    }
}

impl From<&SimpleIfEnumIfElseSimpleEnum> for SimpleEnum {
    fn from(v: &SimpleIfEnumIfElseSimpleEnum) -> Self {
        match &v {
            SimpleIfEnumIfElseSimpleEnum::ZERO { .. } => Self::ZERO,
            SimpleIfEnumIfElseSimpleEnum::ONE { .. } => Self::ONE,
            SimpleIfEnumIfElseSimpleEnum::TWO { .. } => Self::TWO,
        }
    }
}

impl Default for SimpleIfEnumIfElseSimpleEnum {
    fn default() -> Self {
        // First enumerator without any fields
        Self::ZERO {
            basic: Default::default(),
        }
    }
}

impl SimpleIfEnumIfElseSimpleEnum {
    pub fn write<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write(w)?;
        Ok(())
    }

    pub fn write_u16_le<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u16_le(w)
    }

    pub fn write_u16_be<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u16_be(w)
    }

    pub fn write_u32_le<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u32_le(w)
    }

    pub fn write_u32_be<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u32_be(w)
    }

    pub fn write_u64_le<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u64_le(w)
    }

    pub fn write_u64_be<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        let a: SimpleEnum = self.into();
        a.write_u64_be(w)
    }

}

impl VariableSized for SimpleIfEnumIfElseSimpleEnum {
    fn size(&self) -> usize {
        match self {
            Self::ZERO  {
                basic,
            } => {
                1
                + 1 // basic: u8
            }
            Self::ONE  {
                basic,
            } => {
                1
                + 1 // basic: u8
            }
            Self::TWO  {
                other,
            } => {
                1
                + 2 // other: u16
            }
        }
    }
}

impl MaximumPossibleSized for SimpleIfEnumIfElseSimpleEnum {
    fn maximum_possible_size() -> usize {
        65536 // maximum possible u16 size. TODO value.
    }
}
