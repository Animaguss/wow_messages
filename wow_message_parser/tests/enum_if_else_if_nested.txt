#[derive(Debug, PartialEq, Clone, Default)]
pub struct EnumIfElseIfNested {
    pub e: EnumIfElseIfNestedSimpleEnum,
    pub n: EnumIfElseIfNestedNestedEnum,
}

impl ReadableAndWritable for EnumIfElseIfNested {
    type Error = EnumIfElseIfNestedError;

    fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, Self::Error> {
        // e: SimpleEnum
        let e = SimpleEnum::read(r)?;

        // n: NestedEnum
        let n = NestedEnum::read(r)?;

        let e_if = match e {
            SimpleEnum::ZERO => {
                // zero: u8
                let zero = crate::util::read_u8_le(r)?;

                EnumIfElseIfNestedSimpleEnum::ZERO {
                    zero,
                }
            }
            SimpleEnum::ONE => {
                let n_if = match n {
                    NestedEnum::NESTED => {
                        // nested: u8
                        let nested = crate::util::read_u8_le(r)?;

                        EnumIfElseIfNestedNestedEnum::NESTED {
                            nested,
                        }
                    }
                    NestedEnum::NOT_NESTED => EnumIfElseIfNestedNestedEnum::NOT_NESTED,
                };

                EnumIfElseIfNestedSimpleEnum::ONE {
                    n: n_if,
                }
            }
            SimpleEnum::TWO => {
                let n_if = match n {
                    NestedEnum::NESTED => {
                        // nested: u8
                        let nested = crate::util::read_u8_le(r)?;

                        EnumIfElseIfNestedNestedEnum::NESTED {
                            nested,
                        }
                    }
                    NestedEnum::NOT_NESTED => EnumIfElseIfNestedNestedEnum::NOT_NESTED,
                };

                EnumIfElseIfNestedSimpleEnum::TWO {
                    n: n_if,
                }
            }
        };

        Ok(Self {
            e: e_if,
            n: n_if,
        })
    }

    fn write<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // e: SimpleEnum
        self.e.write(w)?;

        // n: NestedEnum
        self.n.write(w)?;

        match &self.e {
            EnumIfElseIfNestedSimpleEnum::ZERO {
                zero,
            } => {
                // zero: u8
                w.write_all(&zero.to_le_bytes())?;

            }
            EnumIfElseIfNestedSimpleEnum::ONE {
            } => {
                match &self.n {
                    EnumIfElseIfNestedNestedEnum::NESTED {
                        nested,
                    } => {
                        // nested: u8
                        w.write_all(&nested.to_le_bytes())?;

                    }
                    EnumIfElseIfNestedNestedEnum::NOT_NESTED => {}
                }

            }
            EnumIfElseIfNestedSimpleEnum::TWO {
            } => {
                match &self.n {
                    EnumIfElseIfNestedNestedEnum::NESTED {
                        nested,
                    } => {
                        // nested: u8
                        w.write_all(&nested.to_le_bytes())?;

                    }
                    EnumIfElseIfNestedNestedEnum::NOT_NESTED => {}
                }

            }
        }

        Ok(())
    }

}
