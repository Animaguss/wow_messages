use std::convert::{TryFrom, TryInto};
use crate::world::v1::v12::{SimpleEnum, SimpleEnumError};
use crate::{ConstantSized, MaximumPossibleSized, VariableSized};
#[cfg(feature = "async_tokio")]
use tokio::io::{AsyncReadExt, AsyncWriteExt};
#[cfg(feature = "async_std")]
use async_std::io::{ReadExt, WriteExt};

#[derive(Debug, PartialEq, Clone, Default)]
pub struct SimpleIfEnumNot {
    pub enum: SimpleIfEnumNotSimpleEnum,
}

impl SimpleIfEnumNot {
    pub(crate) fn read<R: std::io::Read>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumNotError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::read_u8_le(r)?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => SimpleIfEnumNotSimpleEnum::ZERO,
            SimpleEnum::ONE => {
                // others: u8
                let others = crate::util::read_u8_le(r)?;

                SimpleIfEnumNotSimpleEnum::ONE {
                    others,
                }
            }
            SimpleEnum::TWO => {
                // others: u8
                let others = crate::util::read_u8_le(r)?;

                SimpleIfEnumNotSimpleEnum::TWO {
                    others,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    pub(crate) fn write<W: std::io::Write>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        w.write_all(&(self.enum.as_int() as u8).to_le_bytes())?;

        match &self.enum {
            SimpleIfEnumNotSimpleEnum::ZERO => {}
            SimpleIfEnumNotSimpleEnum::ONE {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes())?;

            }
            SimpleIfEnumNotSimpleEnum::TWO {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes())?;

            }
        }

        Ok(())
    }

    pub(crate) async fn tokio_read<R: AsyncReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumNotError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::tokio_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => SimpleIfEnumNotSimpleEnum::ZERO,
            SimpleEnum::ONE => {
                // others: u8
                let others = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumNotSimpleEnum::ONE {
                    others,
                }
            }
            SimpleEnum::TWO => {
                // others: u8
                let others = crate::util::tokio_read_u8_le(r).await?;

                SimpleIfEnumNotSimpleEnum::TWO {
                    others,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    pub(crate) async fn tokio_write<W: AsyncWriteExt + Unpin + Send>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        w.write_all(&(self.enum.as_int() as u8).to_le_bytes()).await?;

        match &self.enum {
            SimpleIfEnumNotSimpleEnum::ZERO => {}
            SimpleIfEnumNotSimpleEnum::ONE {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes()).await?;

            }
            SimpleIfEnumNotSimpleEnum::TWO {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes()).await?;

            }
        }

        Ok(())
    }

    pub(crate) async fn astd_read<R: ReadExt + Unpin + Send>(r: &mut R) -> std::result::Result<Self, SimpleIfEnumNotError> {
        // enum: SimpleEnum
        let enum: SimpleEnum = crate::util::astd_read_u8_le(r).await?.try_into()?;

        let enum_if = match enum {
            SimpleEnum::ZERO => SimpleIfEnumNotSimpleEnum::ZERO,
            SimpleEnum::ONE => {
                // others: u8
                let others = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumNotSimpleEnum::ONE {
                    others,
                }
            }
            SimpleEnum::TWO => {
                // others: u8
                let others = crate::util::astd_read_u8_le(r).await?;

                SimpleIfEnumNotSimpleEnum::TWO {
                    others,
                }
            }
        };

        Ok(Self {
            enum: enum_if,
        })
    }

    pub(crate) async fn astd_write<W: WriteExt + Unpin + Send>(&self, w: &mut W) -> std::result::Result<(), std::io::Error> {
        // enum: SimpleEnum
        w.write_all(&(self.enum.as_int() as u8).to_le_bytes()).await?;

        match &self.enum {
            SimpleIfEnumNotSimpleEnum::ZERO => {}
            SimpleIfEnumNotSimpleEnum::ONE {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes()).await?;

            }
            SimpleIfEnumNotSimpleEnum::TWO {
                others,
            } => {
                // others: u8
                w.write_all(&others.to_le_bytes()).await?;

            }
        }

        Ok(())
    }

}

impl VariableSized for SimpleIfEnumNot {
    fn size(&self) -> usize {
        0
        + self.enum.size() // enum: SimpleIfEnumNotSimpleEnum
    }
}

impl MaximumPossibleSized for SimpleIfEnumNot {
    fn maximum_possible_size() -> usize {
        0
        + 2 // enum: SimpleIfEnumNotSimpleEnum
    }
}

#[derive(Debug)]
pub enum SimpleIfEnumNotError {
    Io(std::io::Error),
    SimpleEnum(SimpleEnumError),
}

impl std::error::Error for SimpleIfEnumNotError {}
impl std::fmt::Display for SimpleIfEnumNotError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Io(i) => i.fmt(f),
            Self::SimpleEnum(i) => i.fmt(f),
        }
    }
}

impl From<std::io::Error> for SimpleIfEnumNotError {
    fn from(e : std::io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<SimpleEnumError> for SimpleIfEnumNotError {
    fn from(e: SimpleEnumError) -> Self {
        Self::SimpleEnum(e)
    }
}

#[derive(Debug, PartialEq, Clone)]
pub enum SimpleIfEnumNotSimpleEnum {
    ZERO,
    ONE {
        others: u8,
    },
    TWO {
        others: u8,
    },
}

impl Default for SimpleIfEnumNotSimpleEnum {
    fn default() -> Self {
        // First enumerator without any fields
        Self::ZERO
    }
}

impl SimpleIfEnumNotSimpleEnum {
    pub(crate) const fn as_int(&self) -> u8 {
        match self {
            Self::ZERO => 0,
            Self::ONE { .. } => 1,
            Self::TWO { .. } => 2,
        }
    }

}

impl VariableSized for SimpleIfEnumNotSimpleEnum {
    fn size(&self) -> usize {
        match self {
            Self::ZERO => {
                1
            }
            Self::ONE {
                others,
            } => {
                1
                + 1 // others: u8
            }
            Self::TWO {
                others,
            } => {
                1
                + 1 // others: u8
            }
        }
    }
}

impl MaximumPossibleSized for SimpleIfEnumNotSimpleEnum {
    fn maximum_possible_size() -> usize {
        2
    }
}
